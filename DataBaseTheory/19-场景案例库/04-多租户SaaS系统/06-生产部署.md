### 2.1 StatefulSet配置

```yaml
# postgresql-statefulset.yaml
apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: postgres-saas
spec:
  serviceName: postgres-saas
  replicas: 3
  selector:
    matchLabels:
      app: postgres-saas
  template:
    metadata:
      labels:
        app: postgres-saas
    spec:
      containers:
      - name: postgres
        image: postgres:18
        env:
        - name: POSTGRES_PASSWORD
          valueFrom:
            secretKeyRef:
              name: postgres-secret
              key: password
        - name: POSTGRES_DB
          value: "saas_db"

        # PostgreSQL 18优化配置
        - name: POSTGRES_INITDB_ARGS
          value: "--encoding=UTF8 --locale=en_US.UTF-8"

        ports:
        - containerPort: 5432
          name: postgres

        volumeMounts:
        - name: data
          mountPath: /var/lib/postgresql/data
        - name: config
          mountPath: /etc/postgresql

        resources:
          requests:
            memory: "8Gi"
            cpu: "2"
          limits:
            memory: "16Gi"
            cpu: "4"

        livenessProbe:
          exec:
            command:
            - pg_isready
            - -U
            - postgres
          initialDelaySeconds: 30
          periodSeconds: 10

        readinessProbe:
          exec:
            command:
            - pg_isready
            - -U
            - postgres
          initialDelaySeconds: 5
          periodSeconds: 5

  volumeClaimTemplates:
  - metadata:
      name: data
    spec:
      accessModes: ["ReadWriteOnce"]
      storageClassName: fast-ssd
      resources:
        requests:
          storage: 500Gi
```

## 4. 初始化脚本

### 4.1 租户初始化

```sql
-- init-tenant.sql
-- 为新租户初始化schema和数据

CREATE OR REPLACE FUNCTION init_tenant(tenant_name TEXT)
RETURNS VOID AS $$
DECLARE
    tenant_schema TEXT;
BEGIN
    tenant_schema := 'tenant_' || tenant_name;

    -- 创建schema
    EXECUTE format('CREATE SCHEMA IF NOT EXISTS %I', tenant_schema);

    -- 创建表
    EXECUTE format('CREATE TABLE %I.users (
        id BIGSERIAL PRIMARY KEY,
        username VARCHAR(100),
        email VARCHAR(255),
        created_at TIMESTAMPTZ DEFAULT now()
    )', tenant_schema);

    EXECUTE format('CREATE TABLE %I.data (
        id BIGSERIAL PRIMARY KEY,
        user_id BIGINT,
        content JSONB,
        created_at TIMESTAMPTZ DEFAULT now()
    )', tenant_schema);

    -- 创建索引
    EXECUTE format('CREATE INDEX idx_%I_users_email ON %I.users(email)',
                   tenant_name, tenant_schema);
    EXECUTE format('CREATE INDEX idx_%I_data_user ON %I.data(user_id)',
                   tenant_name, tenant_schema);

    -- 创建角色
    EXECUTE format('CREATE ROLE %I LOGIN PASSWORD %L',
                   tenant_name, md5(random()::TEXT));
    EXECUTE format('GRANT USAGE ON SCHEMA %I TO %I',
                   tenant_schema, tenant_name);
    EXECUTE format('GRANT ALL ON ALL TABLES IN SCHEMA %I TO %I',
                   tenant_schema, tenant_name);

    RAISE NOTICE '租户 % 初始化完成', tenant_name;
END;
$$ LANGUAGE plpgsql;

-- 使用
SELECT init_tenant('company_abc');
SELECT init_tenant('company_xyz');
```

```yaml
    rules:
      # 租户连接数过多

          pg_stat_activity_count{tenant="{{$labels.tenant}}"} > 100
        for: 5m

      # 租户存储超限
      - alert: TenantStorageLimit

          summary: "租户{{ $labels.tenant }}存储超过10GB"

      - alert: TenantQuerySlow

        for: 10m
        annotations:
          summary: "租户{{ $labels.tenant }}查询变慢"
```

## 6. 备份策略

### 6.1 租户级别备份

```bash
#!/bin/bash
# backup-tenant.sh - 备份单个租户

TENANT_NAME=$1
BACKUP_DIR=/backup/tenants

if [ -z "$TENANT_NAME" ]; then
    echo "用法: $0 <tenant_name>"
    exit 1
fi

SCHEMA="tenant_${TENANT_NAME}"
BACKUP_FILE="${BACKUP_DIR}/${TENANT_NAME}_$(date +%Y%m%d_%H%M%S).dump"

# 备份租户schema
pg_dump \
    -h localhost \
    -U postgres \
    -n $SCHEMA \
    -Fc \
    -f $BACKUP_FILE \
    saas_db

echo "租户 $TENANT_NAME 备份完成: $BACKUP_FILE"

# 保留最近30天备份
find $BACKUP_DIR -name "${TENANT_NAME}_*.dump" -mtime +30 -delete
```

### 6.2 定时备份

```bash
# crontab
```

## 8. 灾难恢复

### 8.1 租户恢复

```bash
#!/bin/bash
# restore-tenant.sh - 恢复单个租户

TENANT_NAME=$1
BACKUP_FILE=$2

if [ -z "$BACKUP_FILE" ]; then
    echo "用法: $0 <tenant_name> <backup_file>"
    exit 1
fi

SCHEMA="tenant_${TENANT_NAME}"

# 1. 删除旧schema（谨慎！）
psql -c "DROP SCHEMA IF EXISTS $SCHEMA CASCADE;"

# 2. 恢复备份
pg_restore -n $SCHEMA -d saas_db $BACKUP_FILE

# 3. 验证
psql -c "SELECT COUNT(*) FROM ${SCHEMA}.users;"

echo "租户 $TENANT_NAME 恢复完成"
```

```sql
-- 1. 确保tenant_id有索引
CREATE INDEX idx_users_tenant ON users(tenant_id);
CREATE INDEX idx_data_tenant ON data(tenant_id);

-- 2. 分区表（按租户）
CREATE TABLE data_partitioned (
    id BIGSERIAL,
    tenant_id INT NOT NULL,
    content JSONB,
    created_at TIMESTAMPTZ
) PARTITION BY HASH (tenant_id);

-- 创建32个分区
DO $$
BEGIN
    FOR i IN 0..31 LOOP
        EXECUTE format('CREATE TABLE data_part_%s PARTITION OF data_partitioned FOR VALUES WITH (MODULUS 32, REMAINDER %s)', i, i);
    END LOOP;
END $$;
```

## 10. 成本优化

### 10.1 存储分层

```sql
-- 热数据（SSD，近3个月）
CREATE TABLESPACE hot_data
    LOCATION '/mnt/ssd/pgdata';

-- 温数据（SSD，3-12个月）
CREATE TABLESPACE warm_data
    LOCATION '/mnt/ssd2/pgdata';

-- 冷数据（HDD，12个月前）
CREATE TABLESPACE cold_data
    LOCATION '/mnt/hdd/pgdata';

-- 自动分层（定时任务）
-- 每月1日执行
DO $$
DECLARE
    tenant_rec RECORD;
BEGIN
    FOR tenant_rec IN
        SELECT DISTINCT tenant_id FROM users
    LOOP
        -- 移动冷数据到HDD
        EXECUTE format('ALTER TABLE tenant_%s.data_archive SET TABLESPACE cold_data', tenant_rec.tenant_id);
    END LOOP;
END $$;
```

## 11. 安全加固

### 11.1 租户隔离验证

```python
def test_tenant_isolation():
    """测试租户隔离"""

    # 租户A连接
    conn_a = psycopg2.connect(
        "dbname=saas_db user=tenant_a password=xxx"
    )
    cursor_a = conn_a.cursor()

    # 设置租户上下文
    cursor_a.execute("SET app.tenant_id = 1")

    # 尝试访问租户B的数据
    try:
        cursor_a.execute("SELECT * FROM users WHERE tenant_id = 2")
        results = cursor_a.fetchall()

        if len(results) > 0:
            print("❌ 租户隔离失败！租户A可以访问租户B数据")
            return False
        else:
            print("✅ 租户隔离正常")
            return True

    except Exception as e:
        print(f"✅ 租户隔离正常（策略阻止）: {e}")
        return True
```
