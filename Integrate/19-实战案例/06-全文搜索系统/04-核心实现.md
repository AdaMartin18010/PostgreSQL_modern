```python
"""
全文搜索引擎
"""

import psycopg2
from psycopg2.extras import RealDictCursor
from fastapi import FastAPI, Query
import time

app = FastAPI()

DB_CONFIG = {
    'dbname': 'search_db',
    'user': 'postgres',
    'host': 'localhost'
}

def get_connection():
    """获取数据库连接（带完整错误处理）"""
    try:
        return psycopg2.connect(**DB_CONFIG, cursor_factory=RealDictCursor)
    except psycopg2.OperationalError as e:
        print(f"❌ 数据库连接失败: {e}")
        raise
    except psycopg2.Error as e:
        print(f"❌ 数据库初始化失败: {e}")
        raise
    except Exception as e:
        print(f"❌ 未知错误: {e}")
        raise

@app.get("/api/search")
async def search(
    q: str = Query(..., description="搜索关键词"),
    category: str = Query(None, description="分类过滤"),
    limit: int = Query(20, le=100),
    offset: int = Query(0, ge=0)
):
    """
    全文搜索API

    支持:
    - 中文分词
    - 相关性排序
    - 高亮显示
    - 分类过滤
    """

    # 参数验证
    if not q or not q.strip():
        from fastapi import HTTPException
        raise HTTPException(status_code=400, detail="搜索关键词不能为空")
    if limit <= 0 or limit > 1000:
        limit = 20
    if offset < 0:
        offset = 0

    start_time = time.time()

    conn = None
    cursor = None
    try:
        conn = get_connection()
        cursor = conn.cursor()

        # 构建查询
        if category:
            query = """
                SELECT
                    doc_id,
                    title,
                    ts_headline('zh_parser', content, query,
                        'MaxWords=50, MinWords=30, StartSel=<b>, StopSel=</b>') AS snippet,
                    ts_rank_cd(search_vector, query) AS rank
                FROM documents,
                     to_tsquery('zh_parser', %s) query
                WHERE search_vector @@ query
                  AND category = %s
                ORDER BY rank DESC, created_at DESC
                LIMIT %s OFFSET %s;
            """
            cursor.execute(query, (q, category, limit, offset))
        else:
            query = """
                SELECT
                    doc_id,
                    title,
                    ts_headline('zh_parser', content, query,
                        'MaxWords=50, MinWords=30, StartSel=<b>, StopSel=</b>') AS snippet,
                    ts_rank_cd(search_vector, query) AS rank
                FROM documents,
                     to_tsquery('zh_parser', %s) query
                WHERE search_vector @@ query
                ORDER BY rank DESC, created_at DESC
                LIMIT %s OFFSET %s;
            """
            cursor.execute(query, (q, limit, offset))

        results = cursor.fetchall()
        duration = (time.time() - start_time) * 1000

        # 记录搜索日志
        try:
            cursor.execute("""
                INSERT INTO search_logs (query_text, result_count, duration_ms)
                VALUES (%s, %s, %s);
            """, (q, len(results), duration))
            conn.commit()
        except Exception as e:
            print(f"记录搜索日志失败: {e}")
            conn.rollback()

        return {
            'success': True,
            'results': results,
            'count': len(results),
            'duration_ms': duration
        }

    except psycopg2.OperationalError as e:
        print(f"数据库连接失败: {e}")
        raise HTTPException(status_code=503, detail="数据库连接失败")
    except psycopg2.Error as e:
        print(f"数据库操作失败: {e}")
        if conn:
            conn.rollback()
        raise HTTPException(status_code=500, detail="数据库操作失败")
    except Exception as e:
        print(f"未知错误: {e}")
        raise HTTPException(status_code=500, detail="系统错误")
    finally:
        if cursor:
            cursor.close()
        if conn:
            conn.close()
```

```python
class SearchIndexManager:
    """搜索索引管理器"""

    def __init__(self, conn):
        self.conn = conn
        self.cursor = conn.cursor()

    def rebuild_search_index(self):
        """重建搜索索引（带完整错误处理）"""
        try:
            # 检查表是否存在
            self.cursor.execute("""
                SELECT 1 FROM information_schema.tables
                WHERE table_schema = 'public' AND table_name = 'documents';
            """)
            if not self.cursor.fetchone():
                raise ValueError("表 documents 不存在")

            # 1. 删除旧索引
            self.cursor.execute("DROP INDEX IF EXISTS idx_documents_search;")

            # 2. 重新创建（PostgreSQL 18并行构建）
            self.cursor.execute("""
                CREATE INDEX idx_documents_search
                ON documents
                USING GIN (search_vector);
            """)

            self.conn.commit()
            print("✅ 搜索索引重建完成")
        except psycopg2.ProgrammingError as e:
            print(f"❌ 重建索引失败（语法错误）: {e}")
            self.conn.rollback()
            raise
        except psycopg2.OperationalError as e:
            print(f"❌ 数据库连接错误: {e}")
            self.conn.rollback()
            raise
        except psycopg2.Error as e:
            print(f"❌ 重建索引失败: {e}")
            self.conn.rollback()
            raise
        except Exception as e:
            print(f"❌ 未知错误: {e}")
            self.conn.rollback()
            raise

    def get_index_stats(self):
        """获取索引统计（带完整错误处理）"""

        except psycopg2.OperationalError as e:
            print(f"❌ 数据库连接错误: {e}")
```

```sql
-- 搜索日志统计函数（带完整错误处理）
CREATE OR REPLACE FUNCTION get_search_statistics(
    p_start_date TIMESTAMPTZ DEFAULT NOW() - INTERVAL '7 days',
    p_end_date TIMESTAMPTZ DEFAULT NOW()
)
RETURNS TABLE (
    metric_name TEXT,
    metric_value NUMERIC,
    description TEXT
) AS $$
BEGIN
    BEGIN
        IF p_start_date IS NULL OR p_end_date IS NULL THEN
            RAISE EXCEPTION 'start_date和end_date不能为NULL';
        END IF;

        IF p_start_date > p_end_date THEN
            RAISE EXCEPTION 'start_date不能大于end_date';
        END IF;

        IF NOT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'search_logs') THEN
            RAISE EXCEPTION '表 search_logs 不存在';
        END IF;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '参数验证失败: %', SQLERRM;
            RAISE;
    END;

    BEGIN
        -- 总搜索次数
        RETURN QUERY SELECT
            '总搜索次数'::TEXT,
            COUNT(*)::NUMERIC,
            '时间段内的总搜索次数'::TEXT
        FROM search_logs
        WHERE created_at BETWEEN p_start_date AND p_end_date;

        -- 平均结果数
        RETURN QUERY SELECT
            '平均结果数'::TEXT,
            ROUND(AVG(result_count), 2)::NUMERIC,
            '平均每次搜索返回的结果数'::TEXT
        FROM search_logs
        WHERE created_at BETWEEN p_start_date AND p_end_date;

        -- 平均响应时间
        RETURN QUERY SELECT
            '平均响应时间(ms)'::TEXT,
            ROUND(AVG(duration_ms), 2)::NUMERIC,
            '平均搜索响应时间'::TEXT
        FROM search_logs
        WHERE created_at BETWEEN p_start_date AND p_end_date;

        -- 零结果率
        RETURN QUERY SELECT
            '零结果率(%)'::TEXT,
            ROUND(COUNT(*) FILTER (WHERE result_count = 0) * 100.0 / NULLIF(COUNT(*), 0), 2)::NUMERIC,
            '没有返回结果的搜索占比'::TEXT
        FROM search_logs
        WHERE created_at BETWEEN p_start_date AND p_end_date;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '搜索统计失败: %', SQLERRM;
            RAISE;
    END;
END;
$$ LANGUAGE plpgsql;
```

```sql
    p_days INT DEFAULT 7

RETURNS TABLE (
    query_text TEXT,
    search_count BIGINT,
    avg_result_count NUMERIC,
    avg_duration_ms NUMERIC

BEGIN
    BEGIN
        -- 参数验证
        IF p_limit IS NULL OR p_limit <= 0 THEN
            RAISE EXCEPTION 'limit必须大于0';
        END IF;

        IF p_days IS NULL OR p_days <= 0 THEN
            RAISE EXCEPTION 'days必须大于0';

            RAISE EXCEPTION '表 search_logs 不存在';
        END IF;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '参数验证失败: %', SQLERRM;
            RAISE;

    BEGIN
        RETURN QUERY
        SELECT
            query_text,
            COUNT(*)::BIGINT AS search_count,
            ROUND(AVG(result_count), 2) AS avg_result_count,
            ROUND(AVG(duration_ms), 2) AS avg_duration_ms

        ORDER BY search_count DESC

        WHEN OTHERS THEN
            RAISE WARNING '热门搜索词查询失败: %', SQLERRM;
            RAISE;
```

```python
class SearchPerformanceMonitor:
    """搜索性能监控"""

    def __init__(self, conn_str):
        self.conn = psycopg2.connect(conn_str)
        self.cursor = self.conn.cursor()

    def log_search(self, query: str, result_count: int, duration_ms: float):
        """记录搜索日志"""
        self.cursor.execute("""
            INSERT INTO search_logs (
                query_text, result_count, duration_ms, created_at
            ) VALUES (%s, %s, %s, NOW())
        """, (query[:200], result_count, duration_ms))
        self.conn.commit()

    def get_search_stats(self, hours: int = 24):
        """获取搜索统计"""
        self.cursor.execute("""
            SELECT
                COUNT(*) AS total_searches,
                AVG(duration_ms) AS avg_duration_ms,
                PERCENTILE_CONT(0.95) WITHIN GROUP (ORDER BY duration_ms) AS p95_duration_ms,
                AVG(result_count) AS avg_result_count
            FROM search_logs
            WHERE created_at > NOW() - INTERVAL '%s hours'
        """, (hours,))

        return self.cursor.fetchone()
```

```python
            INSERT INTO search_clicks (
                query_text, doc_id, click_position, created_at
            ) VALUES (%s, %s, %s, NOW())
        """, (query, clicked_doc_id, 1))

        conn.commit()

        # 计算搜索质量指标
        cursor.execute("""
            SELECT
                COUNT(*) AS total_searches,
                COUNT(*) FILTER (WHERE EXISTS (
                    SELECT 1 FROM search_clicks sc
                    WHERE sc.query_text = sl.query_text
                )) AS searches_with_clicks,
                AVG(result_count) AS avg_result_count
            FROM search_logs sl
            WHERE created_at > NOW() - INTERVAL '7 days'
        """)

        stats = cursor.fetchone()

        click_through_rate = (
            stats['searches_with_clicks'] / stats['total_searches']
            if stats['total_searches'] > 0 else 0
        )

        return {
            'click_through_rate': click_through_rate,
            'avg_result_count': stats['avg_result_count']
        }

    finally:
        cursor.close()
        conn.close()
```
