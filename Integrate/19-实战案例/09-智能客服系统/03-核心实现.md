# 案例9：智能客服系统 - 核心实现

```python
"""
智能客服系统
技术栈: PostgreSQL 18 + pgvector + LangChain
"""

import psycopg2
from psycopg2.extras import RealDictCursor
from langchain.llms import OpenAI
from langchain.embeddings import OpenAIEmbeddings
import numpy as np

class CustomerServiceBot:
    """智能客服机器人"""

    def __init__(self, conn_str, openai_api_key):
        """初始化智能客服机器人（带完整错误处理）"""
        try:
            self.conn = psycopg2.connect(conn_str, cursor_factory=RealDictCursor)
            self.cursor = self.conn.cursor()
        except psycopg2.OperationalError as e:
            raise ConnectionError(f"数据库连接失败: {e}")
        except Exception as e:
            raise RuntimeError(f"初始化失败: {e}")

        try:
            self.llm = OpenAI(api_key=openai_api_key, temperature=0.7)
            self.embeddings = OpenAIEmbeddings(api_key=openai_api_key)
        except Exception as e:
            raise ValueError(f"OpenAI初始化失败: {e}")

    def answer_question(self, session_id: str, question: str) -> dict:
        """
        回答用户问题（带完整错误处理和参数验证）

        流程:
        1. 向量检索相关FAQ
        2. 获取对话历史
        3. LLM生成答案
        4. 保存对话
        """

        # 参数验证
        if not session_id or not session_id.strip():
            raise ValueError("session_id不能为空")
        if not question or not question.strip():
            raise ValueError("question不能为空")

        try:
            # 1. 向量检索
            try:
                question_vec = self.embeddings.embed_query(question)
                relevant_faqs = self._retrieve_faqs(question_vec, top_k=3)
            except Exception as e:
                print(f"向量检索失败: {e}")
                relevant_faqs = []

            # 2. 对话历史
            history = self._get_conversation_history(session_id, limit=5)

            # 3. 生成答案
            answer = self._generate_answer(question, relevant_faqs, history)

            # 4. 保存对话
            self._save_conversation(session_id, question, answer)

            return {
                'session_id': session_id,
                'question': question,
                'answer': answer,
                'sources': [faq['faq_id'] for faq in relevant_faqs] if relevant_faqs else []
            }
        except Exception as e:
            print(f"回答问题失败: {e}")
            return {
                'session_id': session_id,
                'question': question,
                'answer': '抱歉，系统出现错误，请稍后再试。',
                'sources': [],
                'error': str(e)
            }

    def _retrieve_faqs(self, query_vec: list, top_k: int = 3) -> list:
        """向量检索FAQ（带完整错误处理）"""

        try:
            self.cursor.execute("""
                SELECT
                    faq_id,
                    question,
                    answer,
                    category,
                    1 - (embedding <-> %s::vector) AS similarity
                FROM faqs
                WHERE 1 - (embedding <-> %s::vector) > 0.7
                ORDER BY embedding <-> %s::vector
                LIMIT %s;
            """, (query_vec, query_vec, query_vec, top_k))

            return self.cursor.fetchall()
        except psycopg2.OperationalError as e:
            print(f"数据库连接错误: {e}")
            self.conn.rollback()
            return []
        except psycopg2.Error as e:
            print(f"向量检索失败: {e}")
            self.conn.rollback()
            return []
        except Exception as e:
            print(f"未知错误: {e}")
            return []

    def _get_conversation_history(self, session_id: str, limit: int = 5) -> list:
        """获取对话历史（带完整错误处理）"""

        try:
            self.cursor.execute("""
                SELECT question, answer, created_at
                FROM conversations
                WHERE session_id = %s
                ORDER BY created_at DESC
                LIMIT %s;
            """, (session_id, limit))

            return list(reversed(self.cursor.fetchall()))
        except psycopg2.Error as e:
            print(f"获取对话历史失败: {e}")
            return []
        except Exception as e:
            print(f"未知错误: {e}")
            return []

    def _generate_answer(self, question: str, faqs: list, history: list) -> str:
        """LLM生成答案（带完整错误处理）"""

        try:
            # 构建上下文
            context = "\n\n".join([
                f"Q: {faq['question']}\nA: {faq['answer']}"
                for faq in faqs
            ])

            history_text = "\n".join([
                f"用户: {h['question']}\n客服: {h['answer']}"
                for h in history
            ])

            prompt = f"""
你是一个专业的客服助手。基于以下信息回答用户问题。

相关知识:
{context}

对话历史:
{history_text}

用户问题: {question}

请给出专业、友好的回答:
"""

            answer = self.llm(prompt)
            return answer.strip()
        except Exception as e:
            print(f"LLM生成答案失败: {e}")
            return "抱歉，我暂时无法回答您的问题，请稍后再试。"

    def _save_conversation(self, session_id: str, question: str, answer: str):
        """保存对话记录（带完整错误处理）"""

        try:
            self.cursor.execute("""
                INSERT INTO conversations (session_id, question, answer)
                VALUES (%s, %s, %s);
            """, (session_id, question, answer))

            self.conn.commit()
        except psycopg2.IntegrityError as e:
            print(f"保存对话记录失败（数据完整性错误）: {e}")
            self.conn.rollback()
        except psycopg2.OperationalError as e:
            print(f"数据库连接错误: {e}")
            self.conn.rollback()
        except psycopg2.Error as e:
            print(f"保存对话记录失败: {e}")
            self.conn.rollback()
        except Exception as e:
            print(f"未知错误: {e}")
            self.conn.rollback()
```

```python
        self.cursor = conn.cursor()
        self.llm = llm

    def recognize_intent(self, question: str) -> dict:
        """
        识别用户意图

        Returns:
            {
                'intent': 'greeting' | 'question' | 'complaint' | 'other',
                'confidence': 0.0-1.0,
                'entities': [...]
            }
        """

            # 使用LLM识别意图
            prompt = f"""
            识别以下问题的意图：

            问题: {question}

            可能的意图类型：
            - greeting: 问候
            - question: 咨询问题
            - complaint: 投诉
            - feedback: 反馈
            - other: 其他

            请返回JSON格式：
            {{
                "intent": "意图类型",
                "confidence": 0.0-1.0,
                "entities": ["实体1", "实体2"]
            }}
            """

            response = self.llm(prompt)

            # 解析响应（简化处理）
            import json
            result = json.loads(response)

            # 保存意图识别结果
            self.cursor.execute("""
                INSERT INTO intent_logs (question, intent, confidence, entities)
                VALUES (%s, %s, %s, %s::jsonb)
            """, (question, result['intent'], result['confidence'], json.dumps(result['entities'])))

            self.conn.commit()

            return result

        except Exception as e:
            # 记录错误
            self.cursor.execute("""
                INSERT INTO intent_logs (question, intent, confidence, error_message)
                VALUES (%s, 'error', 0.0, %s)
            """, (question, str(e)))
            self.conn.commit()

            return {
                'intent': 'other',
                'confidence': 0.0,
                'entities': []
            }
```

```python
class ConversationManager:
    """对话管理器"""

    def __init__(self, conn):
        self.conn = conn
        self.cursor = conn.cursor()

    def create_session(self, user_id: int = None, channel: str = 'web') -> str:
        """创建新会话"""

        import uuid
        session_id = str(uuid.uuid4())

        try:
            self.cursor.execute("""
                INSERT INTO sessions (session_id, user_id, channel, status)
                VALUES (%s, %s, %s, 'active')
            """, (session_id, user_id, channel))

            self.conn.commit()
            return session_id

        except Exception as e:
            self.conn.rollback()
            raise Exception(f"创建会话失败: {str(e)}")

    def get_session_context(self, session_id: str) -> dict:
        """获取会话上下文"""

        self.cursor.execute("""
            SELECT
                s.session_id,
                s.user_id,
                s.channel,
                s.status,
                COUNT(c.conv_id) AS message_count,
                MAX(c.created_at) AS last_message_time
            FROM sessions s
            LEFT JOIN conversations c ON s.session_id = c.session_id
            WHERE s.session_id = %s
            GROUP BY s.session_id, s.user_id, s.channel, s.status
        """, (session_id,))

        result = self.cursor.fetchone()

        if not result:
            return None

        return {
            'session_id': result['session_id'],
            'user_id': result['user_id'],
            'channel': result['channel'],
            'status': result['status'],
            'message_count': result['message_count'],
            'last_message_time': result['last_message_time']
        }

    def close_session(self, session_id: str):
        """关闭会话"""

        try:
            self.cursor.execute("""
                UPDATE sessions
                SET status = 'closed', updated_at = NOW()
                WHERE session_id = %s
            """, (session_id,))

            self.conn.commit()

        except Exception as e:
            self.conn.rollback()
            raise Exception(f"关闭会话失败: {str(e)}")
```

```python
class FAQManager:
    """FAQ管理器"""

    def __init__(self, conn, embeddings):
        self.conn = conn
        self.cursor = conn.cursor()
        self.embeddings = embeddings

    def add_faq(self, question: str, answer: str, category: str = None):
        """添加FAQ"""

        try:
            # 生成向量
            embedding = self.embeddings.embed_query(question)

            # 插入FAQ
            self.cursor.execute("""
                INSERT INTO faqs (question, answer, category, embedding)
                VALUES (%s, %s, %s, %s::vector)
                RETURNING faq_id
            """, (question, answer, category, embedding))

            faq_id = self.cursor.fetchone()[0]
            self.conn.commit()

            return faq_id

        except Exception as e:
            self.conn.rollback()
            raise Exception(f"添加FAQ失败: {str(e)}")

    def update_faq(self, faq_id: int, question: str = None, answer: str = None):
        """更新FAQ"""

        try:
            updates = []
            params = []

            if question:
                # 重新生成向量
                embedding = self.embeddings.embed_query(question)
                updates.append("question = %s, embedding = %s::vector")
                params.extend([question, embedding])

            if answer:
                updates.append("answer = %s")
                params.append(answer)

            if not updates:
                return

            updates.append("updated_at = NOW()")
            params.append(faq_id)

            query = f"""
                UPDATE faqs
                SET {', '.join(updates)}
                WHERE faq_id = %s
            """

            self.cursor.execute(query, params)
            self.conn.commit()

        except Exception as e:
            self.conn.rollback()
            raise Exception(f"更新FAQ失败: {str(e)}")

    def delete_faq(self, faq_id: int):
        """删除FAQ"""

        try:
            self.cursor.execute("""
                DELETE FROM faqs WHERE faq_id = %s
            """, (faq_id,))

            self.conn.commit()

        except Exception as e:
            self.conn.rollback()
            raise Exception(f"删除FAQ失败: {str(e)}")
```

```sql
-- 意图分布统计函数（带完整错误处理）
CREATE OR REPLACE FUNCTION get_intent_distribution(
    p_start_date TIMESTAMPTZ DEFAULT NOW() - INTERVAL '7 days',
    p_end_date TIMESTAMPTZ DEFAULT NOW()
)
RETURNS TABLE (
    intent TEXT,
    count BIGINT,
    percentage NUMERIC
) AS $$
DECLARE
    total_count BIGINT;
BEGIN
    BEGIN
        IF p_start_date IS NULL OR p_end_date IS NULL THEN
            RAISE EXCEPTION 'start_date和end_date不能为NULL';
        END IF;

        IF p_start_date > p_end_date THEN
            RAISE EXCEPTION 'start_date不能大于end_date';
        END IF;

        IF NOT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'intent_logs') THEN
            RAISE EXCEPTION '表 intent_logs 不存在';
        END IF;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '参数验证失败: %', SQLERRM;
            RAISE;
    END;

    BEGIN
        -- 获取总数
        SELECT COUNT(*) INTO total_count
        FROM intent_logs
        WHERE created_at BETWEEN p_start_date AND p_end_date;

        IF total_count IS NULL THEN
            total_count := 0;
        END IF;

        -- 统计各意图分布
        RETURN QUERY
        SELECT
            il.intent,
            COUNT(*)::BIGINT,
            ROUND(COUNT(*) * 100.0 / NULLIF(total_count, 0), 2)
        FROM intent_logs il
        WHERE il.created_at BETWEEN p_start_date AND p_end_date
        GROUP BY il.intent
        ORDER BY COUNT(*) DESC;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '意图分布统计失败: %', SQLERRM;
            RAISE;
    END;
END;
$$ LANGUAGE plpgsql;
```

```sql
-- 向量检索优化函数（带完整错误处理）
CREATE OR REPLACE FUNCTION optimize_faq_retrieval(
    p_query_text TEXT,
    p_top_k INT DEFAULT 5,
    p_threshold FLOAT DEFAULT 0.7
)
RETURNS TABLE (
    faq_id INT,
    question TEXT,
    answer TEXT,
    similarity FLOAT
) AS $$
DECLARE
    query_vec vector(768);
BEGIN
    BEGIN
        IF p_query_text IS NULL OR p_query_text = '' THEN
            RAISE EXCEPTION 'query_text不能为空';
        END IF;

        IF p_top_k IS NULL OR p_top_k <= 0 THEN
            RAISE EXCEPTION 'top_k必须大于0';
        END IF;

        IF NOT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'faqs') THEN
            RAISE EXCEPTION '表 faqs 不存在';
        END IF;

        IF NOT EXISTS (SELECT 1 FROM pg_extension WHERE extname = 'vector') THEN
            RAISE EXCEPTION 'pgvector扩展未安装';
        END IF;

        IF p_threshold IS NULL OR p_threshold < 0 OR p_threshold > 1 THEN

        IF query_vec IS NULL THEN
            RAISE WARNING '查询向量未生成，无法执行检索';
            RETURN;
        END IF;

        -- 向量检索
        RETURN QUERY
        SELECT
            f.faq_id,
            f.question,
            f.answer,
            (1 - (f.embedding <=> query_vec))::FLOAT AS similarity
        FROM faqs f
        WHERE f.embedding <=> query_vec < (1 - p_threshold)
```

**缓存优化实现（带错误处理和性能测试）**：

```python
from functools import lru_cache
import hashlib
import json

class ResponseCache:
    """响应缓存"""

    def __init__(self, maxsize=500):
        self.cache = {}
        self.maxsize = maxsize

    def get_cache_key(self, question: str) -> str:
        """生成缓存键"""
        return hashlib.md5(question.encode()).hexdigest()

    def get(self, question: str) -> dict:
        """获取缓存"""
        cache_key = self.get_cache_key(question)
        return self.cache.get(cache_key)

    def set(self, question: str, response: dict):
        """设置缓存"""
        cache_key = self.get_cache_key(question)

        # LRU淘汰
        if len(self.cache) >= self.maxsize:
            # 删除最旧的项（简化处理）
            oldest_key = next(iter(self.cache))
            del self.cache[oldest_key]

        self.cache[cache_key] = response

    def clear(self):
        """清空缓存"""
        self.cache.clear()
```
