```python
"""

import psycopg2
from fastapi import FastAPI, Depends, HTTPException
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
import jwt
from datetime import datetime, timedelta

app = FastAPI()
security = HTTPBearer()

# JWT配置
JWT_SECRET = "your-secret-key"
JWT_ALGORITHM = "HS256"

class TenantManager:
    """租户管理器"""

    def __init__(self, conn):
        self.conn = conn
        self.cursor = conn.cursor()

    def create_tenant(self, tenant_name, plan_type='free'):
        """
        创建新租户（带完整错误处理和参数验证）

        Args:
            tenant_name: 租户名称
            plan_type: 套餐类型 (free/pro/enterprise)
        """
        import psycopg2

        # 参数验证
        if not tenant_name or not tenant_name.strip():
            raise ValueError("tenant_name不能为空")
        if plan_type not in ['free', 'pro', 'enterprise']:
            raise ValueError(f"plan_type无效: {plan_type}，必须是free/pro/enterprise之一")

        try:
            # 根据套餐设置配额
            quotas = {
                'free': {'storage_gb': 1, 'connections': 5},
                'pro': {'storage_gb': 10, 'connections': 20},
                'enterprise': {'storage_gb': 100, 'connections': 100}
            }

            quota = quotas.get(plan_type, quotas['free'])

            self.cursor.execute("""
                INSERT INTO tenants (tenant_name, plan_type, storage_quota_gb, connection_quota)
                VALUES (%s, %s, %s, %s)
                RETURNING tenant_id;
            """, (tenant_name, plan_type, quota['storage_gb'], quota['connections']))

            tenant_id = self.cursor.fetchone()[0]
            self.conn.commit()

            # 初始化租户数据
            try:
                self._initialize_tenant_data(tenant_id)
            except Exception as e:
                print(f"初始化租户数据失败: {e}")
                # 回滚租户创建
                self.conn.rollback()
                raise

            return tenant_id
        except psycopg2.IntegrityError as e:
            print(f"创建租户失败（数据完整性错误）: {e}")
            self.conn.rollback()
            raise
        except psycopg2.OperationalError as e:
            print(f"数据库连接失败: {e}")
            self.conn.rollback()
            raise
        except psycopg2.Error as e:
            print(f"数据库操作失败: {e}")
            self.conn.rollback()
            raise
        except Exception as e:
            print(f"未知错误: {e}")
            self.conn.rollback()
            raise

    def _initialize_tenant_data(self, tenant_id):
        """初始化租户数据（带完整错误处理和参数验证）"""
        import psycopg2

        # 参数验证
        if tenant_id is None or tenant_id <= 0:
            raise ValueError(f"tenant_id无效: {tenant_id}")

        try:
            # 设置租户上下文（使用参数化查询防止SQL注入）
            self.cursor.execute("SET app.current_tenant = %s;", (tenant_id,))

            # 创建默认管理员用户
            self.cursor.execute("""
                INSERT INTO users (tenant_id, username, email, password_hash)
                VALUES (%s, 'admin', 'admin@example.com', 'hashed_password');
            """, (tenant_id,))

            self.conn.commit()
            print(f"✅ 租户 {tenant_id} 初始化完成")
        except psycopg2.IntegrityError as e:
            print(f"初始化租户数据失败（数据完整性错误）: {e}")
            self.conn.rollback()
            raise
        except psycopg2.Error as e:
            print(f"初始化租户数据失败: {e}")
            self.conn.rollback()
            raise
        except Exception as e:
            print(f"初始化租户数据未知错误: {e}")
```

```python
from contextvars import ContextVar

# 当前租户ID上下文
current_tenant_id = ContextVar('current_tenant_id', default=None)

def get_current_tenant() -> int:
    """获取当前租户ID"""
    tenant_id = current_tenant_id.get()
    if tenant_id is None:
        raise HTTPException(status_code=401, detail="未设置租户上下文")
    return tenant_id

async def set_tenant_context(credentials: HTTPAuthorizationCredentials = Depends(security)):
    """
    从JWT Token解析租户ID并设置上下文
    """

    try:
        token = credentials.credentials
        payload = jwt.decode(token, JWT_SECRET, algorithms=[JWT_ALGORITHM])
        tenant_id = payload.get('tenant_id')

        if not tenant_id:
            raise HTTPException(status_code=401, detail="Token中无租户信息")

        # 设置上下文
        current_tenant_id.set(tenant_id)

        return tenant_id

    except jwt.JWTError:
        raise HTTPException(status_code=401, detail="Token无效")

# 数据库连接 + 设置RLS上下文（带完整错误处理）
def get_db_with_tenant():
    """获取带租户上下文的数据库连接（带完整错误处理）"""
    import psycopg2

    try:
        tenant_id = get_current_tenant()
    except HTTPException:
        raise
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"获取租户上下文失败: {str(e)}")

    try:
        conn = psycopg2.connect("dbname=saas_db user=postgres")
        cursor = conn.cursor()

        # 设置租户ID（RLS使用）
        cursor.execute(f"SET app.current_tenant = {tenant_id};")
        conn.commit()

        return conn
    except psycopg2.OperationalError as e:
        raise HTTPException(status_code=503, detail=f"数据库连接失败: {str(e)}")
    except psycopg2.Error as e:
        raise HTTPException(status_code=500, detail=f"数据库错误: {str(e)}")
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"未知错误: {str(e)}")
```

```python
class QuotaChecker:
    """配额检查器"""

    def __init__(self, conn, tenant_id):
        self.conn = conn
        self.cursor = conn.cursor()
        self.tenant_id = tenant_id

    def check_storage_quota(self) -> bool:
        """检查存储配额（带完整错误处理）"""
        import psycopg2

        try:
            self.cursor.execute("""
                SELECT
                    COALESCE(SUM(d.size_bytes), 0) / 1024.0 / 1024.0 / 1024.0 AS used_gb,
                    t.storage_quota_gb
                FROM tenants t
                LEFT JOIN documents d ON t.tenant_id = d.tenant_id
                WHERE t.tenant_id = %s
            GROUP BY t.storage_quota_gb;
        """, (self.tenant_id,))

        result = self.cursor.fetchone()
        if not result:
            return True

        used_gb, quota_gb = result
        return used_gb < quota_gb

    def get_quota_usage(self):
        """获取配额使用情况（带完整错误处理）"""
        import psycopg2

        try:
            self.cursor.execute("""
                SELECT
                    t.storage_quota_gb,
                    COALESCE(SUM(d.size_bytes), 0) / 1024.0 / 1024.0 / 1024.0 AS used_gb,
                    COUNT(DISTINCT d.id) AS document_count,
                    COUNT(DISTINCT u.id) AS user_count
                FROM tenants t
            LEFT JOIN documents d ON t.tenant_id = d.tenant_id
            LEFT JOIN users u ON t.tenant_id = u.tenant_id
            WHERE t.tenant_id = %s
            GROUP BY t.storage_quota_gb;
        """, (self.tenant_id,))

            result = self.cursor.fetchone()

            if not result:
                return {
                    'quota_gb': 0,
                    'used_gb': 0,
                    'usage_percent': 0,
                    'document_count': 0,
                    'user_count': 0
                }

            return {
                'quota_gb': result[0],
                'used_gb': float(result[1]),
                'usage_percent': (float(result[1]) / result[0] * 100) if result[0] > 0 else 0,
                'document_count': result[2],
                'user_count': result[3]
            }
        except psycopg2.OperationalError as e:
            print(f"数据库连接错误: {e}")
            raise
        except psycopg2.Error as e:
            print(f"获取配额使用情况失败: {e}")
            raise
        except Exception as e:
            print(f"未知错误: {e}")
            raise
```

```python
class TenantDataMigrator:

    def __init__(self, conn):

        self.cursor = conn.cursor()

    def migrate_tenant_data(self, source_tenant_id, target_tenant_id):
        """
        迁移租户数据

        Args:
            source_tenant_id: 源租户ID
            target_tenant_id: 目标租户ID
        """

        try:
            # 1. 迁移用户数据
            self.cursor.execute("""
                INSERT INTO users (tenant_id, username, email, password_hash, created_at)
                SELECT %s, username, email, password_hash, created_at
                FROM users
                WHERE tenant_id = %s
            """, (target_tenant_id, source_tenant_id))

            user_count = self.cursor.rowcount

            # 2. 迁移文档数据
            self.cursor.execute("""
                INSERT INTO documents (tenant_id, user_id, title, content, size_bytes, created_at)
                SELECT %s,
                       (SELECT id FROM users WHERE tenant_id = %s AND username = u.username LIMIT 1),
                       d.title, d.content, d.size_bytes, d.created_at
                FROM documents d
                JOIN users u ON d.user_id = u.id
                WHERE d.tenant_id = %s
            """, (target_tenant_id, target_tenant_id, source_tenant_id))

            doc_count = self.cursor.rowcount

            self.conn.commit()

            return {
                'success': True,
                'users_migrated': user_count,
                'documents_migrated': doc_count
            }

        except Exception as e:
            self.conn.rollback()
            raise Exception(f"数据迁移失败: {str(e)}")
```

```sql
CREATE OR REPLACE FUNCTION optimize_rls_performance()
RETURNS TABLE (
    table_name TEXT,
    rls_enabled BOOLEAN,
    index_exists BOOLEAN,
    recommendation TEXT
) AS $$
BEGIN
    BEGIN
        -- 检查必要的系统表/视图是否存在
        IF NOT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'pg_catalog' AND table_name = 'pg_class') THEN
            RAISE EXCEPTION '系统表 pg_class 不可访问';
        END IF;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '系统检查失败: %', SQLERRM;
            RAISE;
    END;

    BEGIN
        RETURN QUERY
        SELECT
            t.tablename::TEXT,
            (SELECT relrowsecurity FROM pg_class WHERE relname = t.tablename) AS rls_enabled,
            EXISTS (
                SELECT 1 FROM pg_indexes
                WHERE tablename = t.tablename
                  AND indexname LIKE '%tenant%'
            ) AS index_exists,
            CASE
                WHEN NOT EXISTS (
                    SELECT 1 FROM pg_indexes
                    WHERE tablename = t.tablename

                ELSE '索引已存在'
            END
        FROM pg_tables t
        WHERE schemaname = 'public'

    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING 'RLS性能优化检查失败: %', SQLERRM;

END;
```

### 7.2 查询缓存优化

**查询缓存优化函数（带错误处理和性能测试）**：

```python
from functools import lru_cache
import hashlib

class QueryCache:
    """查询缓存"""

    def __init__(self, maxsize=128):
        self.cache = {}
        self.maxsize = maxsize

    def get_cache_key(self, query, tenant_id, params):
        """生成缓存键"""
        cache_str = f"{query}_{tenant_id}_{str(params)}"
        return hashlib.md5(cache_str.encode()).hexdigest()

    def get(self, query, tenant_id, params):
        """获取缓存"""
        cache_key = self.get_cache_key(query, tenant_id, params)
        return self.cache.get(cache_key)

    def set(self, query, tenant_id, params, result):
        """设置缓存"""
        cache_key = self.get_cache_key(query, tenant_id, params)

        # LRU淘汰
        if len(self.cache) >= self.maxsize:
            # 删除最旧的项（简化处理）
            oldest_key = next(iter(self.cache))
            del self.cache[oldest_key]

        self.cache[cache_key] = result

    def clear_tenant_cache(self, tenant_id):
        """清除租户缓存"""
        keys_to_remove = [
            key for key in self.cache.keys()
            if key.startswith(hashlib.md5(str(tenant_id).encode()).hexdigest()[:8])
        ]
        for key in keys_to_remove:
            del self.cache[key]
```

```sql
-- 3. 租户数据清理（定期清理，带完整错误处理）
CREATE OR REPLACE FUNCTION cleanup_tenant_data(
    p_tenant_id INT,
    p_retention_days INT DEFAULT 90
)
RETURNS TABLE (
    deleted_count BIGINT,
    freed_space TEXT
) AS $$
DECLARE
    v_deleted BIGINT;
    v_size_before BIGINT;
    v_size_after BIGINT;
BEGIN
    BEGIN
        -- 参数验证
        IF p_tenant_id IS NULL OR p_tenant_id <= 0 THEN
            RAISE EXCEPTION 'tenant_id必须大于0';
        END IF;

        IF p_retention_days IS NULL OR p_retention_days <= 0 THEN
            RAISE EXCEPTION 'retention_days必须大于0';
        END IF;

        IF NOT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'documents') THEN
            RAISE EXCEPTION '表 documents 不存在';
        END IF;

        IF NOT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'tenants') OR
           NOT EXISTS (SELECT 1 FROM tenants WHERE tenant_id = p_tenant_id) THEN
            RAISE EXCEPTION '租户 % 不存在', p_tenant_id;
        END IF;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '参数验证失败: %', SQLERRM;
            RAISE;
    END;

    BEGIN
        -- 记录删除前大小
        SELECT pg_total_relation_size('documents') INTO v_size_before;

        IF v_size_before IS NULL THEN
            v_size_before := 0;
        END IF;

        RAISE NOTICE '开始清理租户 % 的数据（保留 % 天）', p_tenant_id, p_retention_days;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '记录删除前大小失败: %', SQLERRM;
            RAISE;
    END;

    BEGIN
        -- 删除旧数据
        DELETE FROM documents
        WHERE tenant_id = p_tenant_id
            AND created_at < NOW() - (p_retention_days || ' days')::INTERVAL;

        GET DIAGNOSTICS v_deleted = ROW_COUNT;

        RAISE NOTICE '删除了 % 条记录', v_deleted;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '删除旧数据失败: %', SQLERRM;
            RAISE;
    END;

    BEGIN
        -- 记录删除后大小
        SELECT pg_total_relation_size('documents') INTO v_size_after;

        IF v_size_after IS NULL THEN
            v_size_after := 0;
        END IF;

        -- VACUUM回收空间（注意：VACUUM不能在函数中执行，需要外部调用）
        RAISE NOTICE '建议在函数外部执行 VACUUM documents 以回收空间';
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '记录删除后大小失败: %', SQLERRM;
            RAISE;
    END;

    BEGIN
        RETURN QUERY SELECT
            v_deleted,
            pg_size_pretty(GREATEST(v_size_before - v_size_after, 0));
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '返回清理结果失败: %', SQLERRM;
            RAISE;
    END;
END;
```
