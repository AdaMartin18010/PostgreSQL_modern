"""
金融交易核心模块
特点: ACID保证、乐观锁、完整审计
"""

import psycopg2
from decimal import Decimal
import logging

class TransactionProcessor:
    """交易处理器"""

    def __init__(self, conn_str):
        """初始化交易处理器（带完整错误处理）"""
        try:
            self.conn = psycopg2.connect(conn_str)
            self.cursor = self.conn.cursor()
        except psycopg2.OperationalError as e:
            raise ConnectionError(f"数据库连接失败: {e}")
        except Exception as e:
            raise RuntimeError(f"初始化失败: {e}")

        logging.basicConfig(level=logging.INFO)
        self.logger = logging.getLogger(__name__)

    def transfer(self, from_account, to_account, amount, user_id, max_retries=3):
        """
        转账（带重试和完整参数验证）

        Args:
            from_account: 源账户ID
            to_account: 目标账户ID
            amount: 金额
            user_id: 操作用户ID
            max_retries: 最大重试次数
        """

        # 参数验证
        if from_account is None or from_account <= 0:
            raise ValueError(f"from_account无效: {from_account}")
        if to_account is None or to_account <= 0:
            raise ValueError(f"to_account无效: {to_account}")
        if from_account == to_account:
            raise ValueError("源账户和目标账户不能相同")
        if amount is None or amount <= 0:
            raise ValueError(f"amount必须大于0: {amount}")
        if user_id is None or user_id <= 0:
            raise ValueError(f"user_id无效: {user_id}")
        if max_retries < 1 or max_retries > 10:
            max_retries = 3

        for attempt in range(max_retries):
            try:
                # 调用数据库函数
                self.cursor.execute("""
                    SELECT transfer_money(%s, %s, %s, %s);
                """, (from_account, to_account, amount, user_id))

                transaction_id = self.cursor.fetchone()[0]
                self.conn.commit()

                self.logger.info(f"✅ 转账成功: {transaction_id}")
                return transaction_id

            except psycopg2.Error as e:
                self.conn.rollback()

                # 版本冲突，重试
                if '版本冲突' in str(e):
                    self.logger.warning(f"版本冲突，重试 {attempt + 1}/{max_retries}")
                    continue
                else:
                    # 其他错误，直接抛出
                    self.logger.error(f"❌ 转账失败: {e}")
                    raise

        # 重试耗尽
        raise Exception("转账失败: 达到最大重试次数")

    def get_account_balance(self, account_id):
        """查询账户余额（带完整错误处理和参数验证）"""

        # 参数验证
        if account_id is None or account_id <= 0:
            raise ValueError(f"account_id无效: {account_id}")

        try:
            self.cursor.execute("""
                SELECT balance, currency
                FROM accounts
                WHERE account_id = %s;
            """, (account_id,))

            result = self.cursor.fetchone()
            if not result:
                raise ValueError("账户不存在")

            return {'balance': float(result[0]), 'currency': result[1]}
        except psycopg2.OperationalError as e:
            self.logger.error(f"数据库连接错误: {e}")
            raise
        except psycopg2.Error as e:
            self.logger.error(f"数据库操作失败: {e}")
            raise
        except Exception as e:
            self.logger.error(f"查询账户余额失败: {e}")
            raise

    def get_transaction_history(self, account_id, limit=100):
        """查询交易历史（带完整错误处理和参数验证）"""

        # 参数验证
        if account_id is None or account_id <= 0:
            raise ValueError(f"account_id无效: {account_id}")
        if limit <= 0 or limit > 10000:
            limit = 100

        try:
            self.cursor.execute("""
                SELECT
                    transaction_id,
                    CASE
                        WHEN from_account_id = %s THEN 'debit'
                        ELSE 'credit'
                    END AS type,
                    amount,
                    CASE
                        WHEN from_account_id = %s THEN to_account_id
                        ELSE from_account_id
                    END AS other_account,
                    status,
                    created_at
                FROM transactions
                WHERE from_account_id = %s OR to_account_id = %s
                ORDER BY created_at DESC
                LIMIT %s;
            """, (account_id, account_id, account_id, account_id, limit))

            return self.cursor.fetchall()
        except psycopg2.OperationalError as e:
            self.logger.error(f"数据库连接错误: {e}")
            raise
        except psycopg2.Error as e:
            self.logger.error(f"数据库操作失败: {e}")
            raise
        except Exception as e:
            self.logger.error(f"查询交易历史失败: {e}")
            raise

            # 1. 检查账户余额一致性
            inconsistencies = self._check_balance_consistency(date)

            # 2. 检查交易完整性
            incomplete_txs = self._check_transaction_completeness(date)

            if incomplete_txs:
                self.logger.error(f"发现{len(incomplete_txs)}个未完成交易")
                return False

            # 3. 生成对账报告
            report = self._generate_report(date)

            self.logger.info(f"✅ {date}对账完成")
            return True
        except psycopg2.Error as e:
            self.logger.error(f"对账过程数据库错误: {e}")
            self.conn.rollback()
            return False
        except Exception as e:
            self.logger.error(f"对账过程未知错误: {e}")
            return False

    def _check_balance_consistency(self, date):
        """检查余额一致性（带完整错误处理）"""

        try:
            self.cursor.execute("""
                WITH calculated_balance AS (
                    SELECT
                        account_id,
                        COALESCE(SUM(
                            CASE
                                WHEN to_account_id = account_id THEN amount
                                WHEN from_account_id = account_id THEN -amount
                                ELSE 0
                            END
                        ), 0) AS calc_balance
                    FROM accounts a
                    LEFT JOIN transactions t ON (
                        (t.from_account_id = a.account_id OR t.to_account_id = a.account_id)
                        AND t.status = 'success'
                        AND DATE(t.completed_at) <= %s
                    )
                    GROUP BY account_id
                )
                SELECT
                    a.account_id,
                    a.balance AS actual_balance,
                    cb.calc_balance AS calculated_balance,
                    a.balance - cb.calc_balance AS diff
                FROM accounts a
                JOIN calculated_balance cb ON a.account_id = cb.account_id
                WHERE ABS(a.balance - cb.calc_balance) > 0.01;  -- 容差0.01元
            """, (date,))

            return self.cursor.fetchall()
        except psycopg2.Error as e:
            self.logger.error(f"检查余额一致性失败: {e}")
            return []
        except Exception as e:
            self.logger.error(f"未知错误: {e}")
            return []

class EventStore:
    """事件存储（Event Sourcing）"""

    def __init__(self, conn):
        self.conn = conn
        self.cursor = conn.cursor()

    def append_event(self, account_id, event_type, event_data):
        """追加事件"""

        try:
            # 获取下一个序列号
            self.cursor.execute("""
                SELECT COALESCE(MAX(sequence_number), 0) + 1
                FROM account_events
                WHERE account_id = %s;
            """, (account_id,))

            sequence_number = self.cursor.fetchone()[0]

            # 插入事件
            self.cursor.execute("""
                INSERT INTO account_events (
                    account_id, event_type, event_data, sequence_number
                )
                VALUES (%s, %s, %s::jsonb, %s);
            """, (account_id, event_type, json.dumps(event_data), sequence_number))

            self.conn.commit()
            return sequence_number

        except Exception as e:
            self.conn.rollback()
            raise Exception(f"事件追加失败: {e}")

    def replay_events(self, account_id, up_to_sequence=None):
        """重放事件（重建账户状态）"""

        query = """
            SELECT event_type, event_data, sequence_number
            FROM account_events
            WHERE account_id = %s
        """
        params = [account_id]

        if up_to_sequence:
            query += " AND sequence_number <= %s"
            params.append(up_to_sequence)

        query += " ORDER BY sequence_number"

        self.cursor.execute(query, params)
        events = self.cursor.fetchall()

        # 重建账户状态
        balance = Decimal('0.00')
        for event in events:
            event_data = json.loads(event['event_data'])
            if event['event_type'] == 'deposit':
                balance += Decimal(str(event_data['amount']))
            elif event['event_type'] == 'withdraw':
                balance -= Decimal(str(event_data['amount']))
            elif event['event_type'] == 'transfer_out':
                balance -= Decimal(str(event_data['amount']))
            elif event['event_type'] == 'transfer_in':
                balance += Decimal(str(event_data['amount']))

        return balance

class BatchProcessor:
    """批量处理器"""

    def __init__(self, conn, batch_size=1000):
        self.conn = conn
        self.cursor = conn.cursor()
        self.batch_size = batch_size
        self.batch = []

    def add_transaction(self, from_account, to_account, amount, user_id):
        """添加交易到批次"""
        self.batch.append((from_account, to_account, amount, user_id))

        if len(self.batch) >= self.batch_size:
            self.flush()

    def flush(self):
        """刷新批次"""
        if not self.batch:
            return

        try:
            # 使用批量插入
            self.cursor.executemany("""
                SELECT transfer_money(%s, %s, %s, %s);
            """, self.batch)

            self.conn.commit()
            self.batch = []

        except Exception as e:
            self.conn.rollback()
            raise Exception(f"批量处理失败: {e}")

            SELECT
                COUNT(*) AS total_transactions,
                COUNT(*) FILTER (WHERE status = 'success') AS success_count,
                COUNT(*) FILTER (WHERE status = 'failed') AS failed_count,
                AVG(EXTRACT(EPOCH FROM (completed_at - created_at))) AS avg_duration_seconds,
                PERCENTILE_CONT(0.95) WITHIN GROUP (
                    ORDER BY EXTRACT(EPOCH FROM (completed_at - created_at))

            FROM transactions
            WHERE created_at > NOW() - INTERVAL '%s minutes'
        """, (time_window_minutes,))

        """获取账户统计"""

            SELECT
                COUNT(*) AS total_accounts,
                COUNT(*) FILTER (WHERE status = 'active') AS active_accounts,
                SUM(balance) AS total_balance,
                AVG(balance) AS avg_balance
            FROM accounts
        """)

        return self.cursor.fetchone()
