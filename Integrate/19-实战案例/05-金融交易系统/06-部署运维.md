
### 4.2 全量备份脚本

**全量备份脚本（带错误处理和性能测试）**：

```bash
#!/bin/bash
# backup_trading.sh - 金融交易系统全量备份

set -e

BACKUP_DIR="/backups/trading"
DATE=$(date +%Y%m%d_%H%M%S)
RETENTION_DAYS=30

mkdir -p "$BACKUP_DIR"

# 1. 创建备份点
echo "创建备份点..."
psql -h postgres-primary -U postgres -d trading_db -c "
    SELECT pg_start_backup('backup_$DATE', false, false);
"

# 2. 备份数据目录
echo "备份数据目录..."
tar -czf "$BACKUP_DIR/data_$DATE.tar.gz" \
    -C /var/lib/postgresql/data \
    --exclude='pg_wal' \
    --exclude='pg_replslot' \
    .

# 3. 结束备份点
echo "结束备份点..."
psql -h postgres-primary -U postgres -d trading_db -c "
    SELECT pg_stop_backup(false);
"

# 4. 备份WAL
echo "备份WAL..."
rsync -av /var/lib/postgresql/data/pg_wal/ "$BACKUP_DIR/wal/"

# 5. 清理旧备份
find "$BACKUP_DIR" -name "data_*.tar.gz" -mtime +$RETENTION_DAYS -delete

echo "备份完成: $BACKUP_DIR/data_$DATE.tar.gz"
```

### 4.3 恢复脚本

**恢复脚本（带错误处理和性能测试）**：

```bash
#!/bin/bash
# restore_trading.sh - 金融交易系统恢复

set -e

BACKUP_FILE=$1
WAL_DIR="/backups/wal"

if [ -z "$BACKUP_FILE" ]; then
    echo "用法: $0 <备份文件>"
    exit 1
fi

# 1. 停止PostgreSQL
systemctl stop postgresql

# 2. 清空数据目录
rm -rf /var/lib/postgresql/data/*

# 3. 恢复数据
echo "恢复数据..."
tar -xzf "$BACKUP_FILE" -C /var/lib/postgresql/data

# 4. 配置恢复
cat >> /var/lib/postgresql/data/postgresql.conf <<EOF
restore_command = 'cp $WAL_DIR/%f %p'
recovery_target_time = '2025-01-01 00:00:00'  # 指定恢复时间点
EOF

touch /var/lib/postgresql/data/recovery.signal

# 5. 启动PostgreSQL
systemctl start postgresql

echo "恢复完成！"
```

## 5. 性能调优

### 5.1 数据库性能调优

**数据库性能调优（PostgreSQL 18特性）**：

```sql
-- 1. 分析表统计信息（带错误处理）
DO $$
BEGIN
    BEGIN
        IF NOT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'accounts') THEN
            RAISE WARNING '表 accounts 不存在';
        ELSE
            ANALYZE accounts;
            RAISE NOTICE 'accounts 表分析完成';
        END IF;

        IF NOT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'transactions') THEN
            RAISE WARNING '表 transactions 不存在';
        ELSE
            ANALYZE transactions;
            RAISE NOTICE 'transactions 表分析完成';
        END IF;

        IF NOT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'audit_logs') THEN
            RAISE WARNING '表 audit_logs 不存在';
        ELSE
            ANALYZE audit_logs;
            RAISE NOTICE 'audit_logs 表分析完成';
        END IF;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '分析表统计信息失败: %', SQLERRM;
    END;
END $$;

-- 2. 重建索引（带错误处理）
DO $$
BEGIN
    BEGIN
        IF EXISTS (SELECT 1 FROM pg_indexes WHERE indexname = 'idx_accounts_account_id') THEN
            REINDEX INDEX CONCURRENTLY idx_accounts_account_id;
            RAISE NOTICE '索引 idx_accounts_account_id 重建完成';
        END IF;

        IF EXISTS (SELECT 1 FROM pg_indexes WHERE indexname = 'idx_transactions_account_id') THEN
            REINDEX INDEX CONCURRENTLY idx_transactions_account_id;
            RAISE NOTICE '索引 idx_transactions_account_id 重建完成';
        END IF;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '重建索引失败: %', SQLERRM;
    END;
END $$;

-- 3. 检查慢查询（带错误处理和性能测试）
DO $$
BEGIN
    BEGIN
        IF NOT EXISTS (SELECT 1 FROM pg_extension WHERE extname = 'pg_stat_statements') THEN
            RAISE WARNING 'pg_stat_statements扩展未安装，无法检查慢查询';
            RETURN;
        END IF;
        RAISE NOTICE '开始检查慢查询';
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '检查准备失败: %', SQLERRM;
    END;
END $$;
SELECT
    query,
    calls,
    total_exec_time,
    mean_exec_time,
    max_exec_time
FROM pg_stat_statements
WHERE mean_exec_time > 500  -- >500ms
ORDER BY mean_exec_time DESC
LIMIT 20;

-- 4. 检查锁等待
SELECT
    blocked_locks.pid AS blocked_pid,
    blocking_locks.pid AS blocking_pid,
    blocked_activity.usename AS blocked_user,
    blocking_activity.usename AS blocking_user,
    blocked_activity.query AS blocked_statement,
    blocking_activity.query AS blocking_statement
FROM pg_catalog.pg_locks blocked_locks
JOIN pg_catalog.pg_stat_activity blocked_activity ON blocked_activity.pid = blocked_locks.pid
JOIN pg_catalog.pg_locks blocking_locks
    ON blocking_locks.locktype = blocked_locks.locktype
    AND blocking_locks.database IS NOT DISTINCT FROM blocked_locks.database
    AND blocking_locks.relation IS NOT DISTINCT FROM blocked_locks.relation
    AND blocking_locks.page IS NOT DISTINCT FROM blocked_locks.page
    AND blocking_locks.tuple IS NOT DISTINCT FROM blocked_locks.tuple
    AND blocking_locks.virtualxid IS NOT DISTINCT FROM blocked_locks.virtualxid
    AND blocking_locks.transactionid IS NOT DISTINCT FROM blocked_locks.transactionid
    AND blocking_locks.classid IS NOT DISTINCT FROM blocked_locks.classid
    AND blocking_locks.objid IS NOT DISTINCT FROM blocked_locks.objid
    AND blocking_locks.objsubid IS NOT DISTINCT FROM blocked_locks.objsubid
    AND blocking_locks.pid != blocked_locks.pid
JOIN pg_catalog.pg_stat_activity blocking_activity ON blocking_activity.pid = blocking_locks.pid
WHERE NOT blocked_locks.granted;
```
