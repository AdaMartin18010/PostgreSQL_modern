## 2. 基准测试

### 2.1 TPS测试

```python
import psycopg2
import time
from concurrent.futures import ThreadPoolExecutor
import random

def execute_transaction(thread_id, iterations):
    """执行交易（带完整错误处理）"""

    conn = None
    cursor = None
    success_count = 0
    error_count = 0
    total_time = 0

    try:
        conn = psycopg2.connect("dbname=trading_db")
        cursor = conn.cursor()
    except psycopg2.OperationalError as e:
        print(f"线程 {thread_id} 数据库连接失败: {e}")
        return {
            'thread_id': thread_id,
            'success': 0,
            'error': iterations,
            'avg_time': 0
        }

    for i in range(iterations):
        start = time.time()

        try:
            # 随机转账
            from_account = random.randint(1, 100000)
            to_account = random.randint(1, 100000)
            amount = random.uniform(1.0, 1000.0)

            # 调用转账函数
            cursor.execute("""
                SELECT transfer_money(%s, %s, %s)
            """, (from_account, to_account, amount))

            result = cursor.fetchone()[0]

            if result:
                success_count += 1
            else:
                error_count += 1

            conn.commit()

        except psycopg2.IntegrityError as e:
            error_count += 1
            conn.rollback()
        except psycopg2.OperationalError as e:
            print(f"线程 {thread_id} 第 {i+1} 次操作连接失败: {e}")
            error_count += 1
            try:
                conn.rollback()
            except:
                pass
        except psycopg2.Error as e:
            error_count += 1
            conn.rollback()
        except Exception as e:
            print(f"线程 {thread_id} 第 {i+1} 次操作未知错误: {e}")
            error_count += 1
            try:
                conn.rollback()
            except:
                pass

        finally:
            duration = time.time() - start
            total_time += duration

    if cursor:
        cursor.close()
    if conn:
        conn.close()

    return {
        'thread_id': thread_id,
        'success': success_count,
        'error': error_count,
        'avg_time': total_time / iterations * 1000 if iterations > 0 else 0  # ms
    }

# 并发测试（带完整错误处理）
def run_tps_test(threads=50, iterations_per_thread=200):
    """TPS压力测试（带完整错误处理）"""

    print(f"开始TPS测试: {threads}线程 × {iterations_per_thread}次")
    start = time.time()

    try:
        with ThreadPoolExecutor(max_workers=threads) as executor:
            futures = [
                executor.submit(execute_transaction, i, iterations_per_thread)
                for i in range(threads)
            ]

            results = []
            for future in futures:
                try:
                    results.append(future.result())
                except Exception as e:
                    print(f"线程执行失败: {e}")
                    results.append({
                        'thread_id': -1,
                        'success': 0,
                        'error': iterations_per_thread,
                        'avg_time': 0
                    })

        end = time.time()
        total_time = end - start

        # 统计
        total_success = sum(r['success'] for r in results)
        total_error = sum(r['error'] for r in results)
        tps = total_success / total_time if total_time > 0 else 0
        avg_latency = sum(r['avg_time'] for r in results) / len(results) if results else 0
        total_ops = total_success + total_error
        success_rate = total_success * 100 / total_ops if total_ops > 0 else 0

        print(f"\n测试结果:")
        print(f"  总时间: {total_time:.2f}秒")
        print(f"  成功事务: {total_success}")
        print(f"  失败事务: {total_error}")
        print(f"  TPS: {tps:.2f}")
        print(f"  平均延迟: {avg_latency:.2f}ms")
        print(f"  成功率: {success_rate:.2f}%")

        return {
            'total_time': total_time,
            'total_success': total_success,
            'total_error': total_error,
            'tps': tps,
            'avg_latency': avg_latency,
            'success_rate': success_rate
        }
    except Exception as e:
        print(f"TPS测试执行失败: {e}")
        raise

# 运行测试
run_tps_test(threads=50, iterations_per_thread=200)

"""
测试结果:
  总时间: 45.23秒
  成功事务: 9850
  失败事务: 150（余额不足或并发冲突）
  TPS: 217.8
  平均延迟: 22.5ms
  成功率: 98.5%
"""
```

```python
def latency_test(percentiles=[50, 90, 95, 99]):
    """延迟分布测试"""

    conn = psycopg2.connect("dbname=trading_db")
    cursor = conn.cursor()

    latencies = []

    for i in range(10000):
        start = time.time()

        cursor.execute("""
            SELECT transfer_money(%s, %s, %s)
        """, (
            random.randint(1, 100000),
            random.randint(1, 100000),
            random.uniform(1.0, 100.0)
        ))

        conn.commit()

        latency = (time.time() - start) * 1000
        latencies.append(latency)

    cursor.close()
    conn.close()

    # 计算百分位数
    latencies.sort()

    print("延迟分布:")
    for p in percentiles:
        idx = int(len(latencies) * p / 100)
        print(f"  P{p}: {latencies[idx]:.2f}ms")

"""
延迟分布:
  P50: 18.5ms
  P90: 35.2ms
  P95: 48.6ms
  P99: 125.3ms
"""
```

```python
def test_acid_properties():
    """ACID属性验证"""

    conn = psycopg2.connect("dbname=trading_db")
    cursor = conn.cursor()

    # 测试前检查总金额
    cursor.execute("SELECT SUM(balance) FROM accounts")
    total_before = cursor.fetchone()[0]

    # 执行10000笔转账
    for i in range(10000):
        from_acc = random.randint(1, 1000)
        to_acc = random.randint(1, 1000)
        if from_acc == to_acc:
            continue

        try:
            cursor.execute("""
                SELECT transfer_money(%s, %s, %s)
            """, (from_acc, to_acc, 10.0))
            conn.commit()
        except:
            conn.rollback()

    # 测试后检查总金额
    cursor.execute("SELECT SUM(balance) FROM accounts")
    total_after = cursor.fetchone()[0]

    cursor.close()
    conn.close()

    print(f"转账前总金额: {total_before}")
    print(f"转账后总金额: {total_after}")
    print(f"差异: {abs(total_before - total_after)}")

    if abs(total_before - total_after) < 0.01:
        print("✅ ACID一致性验证通过")
    else:
        print("❌ ACID一致性验证失败")
```

### 9.2 性能告警规则

**性能告警规则（带错误处理和性能测试）**：

```sql
-- 性能告警检查函数（带完整错误处理）
CREATE OR REPLACE FUNCTION check_transaction_performance_alerts()
RETURNS TABLE (
    alert_type TEXT,
    alert_message TEXT,
    current_value NUMERIC,
    threshold_value NUMERIC
) AS $$
DECLARE
    v_time_diff NUMERIC;
    v_tps NUMERIC;
BEGIN
    BEGIN
        IF NOT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'transactions') THEN
            RAISE EXCEPTION '表 transactions 不存在';
        END IF;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '表存在性检查失败: %', SQLERRM;
            RAISE;
    END;

    BEGIN
        -- 1. 检查TPS
        SELECT
            EXTRACT(EPOCH FROM (MAX(created_at) - MIN(created_at))) INTO v_time_diff
        FROM transactions
        WHERE created_at > NOW() - INTERVAL '1 minute';

        IF v_time_diff IS NULL OR v_time_diff <= 0 THEN
            v_time_diff := 1;
        END IF;

        SELECT COUNT(*)::NUMERIC / v_time_diff INTO v_tps
        FROM transactions
        WHERE created_at > NOW() - INTERVAL '1 minute';

        IF v_tps IS NOT NULL AND v_tps < 5000.0 THEN
            RETURN QUERY SELECT
                'low_tps'::TEXT,
                'TPS低于阈值'::TEXT,
                v_tps,
                5000.0::NUMERIC;
        END IF;

        -- 2. 检查平均延迟
        RETURN QUERY
        SELECT
            'high_avg_latency'::TEXT,
            '平均处理延迟超过阈值'::TEXT,
            AVG(processing_time_ms)::NUMERIC,
            50.0::NUMERIC
        FROM transactions
        WHERE created_at > NOW() - INTERVAL '1 hour'
        HAVING AVG(processing_time_ms) > 50.0;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '性能告警检查失败: %', SQLERRM;
            RAISE;
    END;
END;
$$ LANGUAGE plpgsql;
```
