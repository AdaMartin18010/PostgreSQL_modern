## 1. 数据导入模块

### 1.1 批量数据导入

```python
"""
OLAP数据ETL模块
用途: 从多源导入数据到PostgreSQL
"""

import psycopg2
from psycopg2.extras import execute_values
import pandas as pd
from datetime import datetime
import logging

class OLAPDataLoader:
    """OLAP数据加载器"""

    def __init__(self, conn_str):
        """初始化OLAP数据加载器（带完整错误处理）"""
        try:
            self.conn = psycopg2.connect(conn_str)
            self.cursor = self.conn.cursor()
        except psycopg2.OperationalError as e:
            raise ConnectionError(f"数据库连接失败: {e}")
        except Exception as e:
            raise RuntimeError(f"初始化失败: {e}")

        logging.basicConfig(level=logging.INFO)
        self.logger = logging.getLogger(__name__)

    def load_fact_table(self, csv_file, table_name, batch_size=10000):
        """
        加载事实表数据（带完整错误处理和参数验证）

        Args:
            csv_file: CSV文件路径
            table_name: 目标表名
            batch_size: 批次大小
        """

        # 参数验证
        if not csv_file or not csv_file.strip():
            raise ValueError("csv_file不能为空")
        if not table_name or not table_name.strip():
            raise ValueError("table_name不能为空")
        if batch_size <= 0 or batch_size > 1000000:
            raise ValueError(f"batch_size必须在1-1000000之间: {batch_size}")

        self.logger.info(f"开始加载数据到 {table_name}")

        # 读取CSV
        try:
            df = pd.read_csv(csv_file)
        except FileNotFoundError:
            raise FileNotFoundError(f"CSV文件不存在: {csv_file}")
        except Exception as e:
            raise ValueError(f"读取CSV文件失败: {e}")
        total_rows = len(df)

        # 批量插入
        inserted = 0
        for i in range(0, total_rows, batch_size):
            batch = df.iloc[i:i+batch_size]

            # 转换为tuple列表
            values = [tuple(row) for row in batch.values]

            # 使用COPY或execute_values
            try:
                # 方法1: COPY（最快）
                self._copy_from_dataframe(batch, table_name)

                # 方法2: execute_values（备选）
                # self._insert_with_execute_values(values, table_name)

                inserted += len(batch)
                self.logger.info(f"已导入 {inserted}/{total_rows} 行")

            except psycopg2.IntegrityError as e:
                self.logger.error(f"批次导入失败（数据完整性错误）: {e}")
                self.conn.rollback()
                raise
            except psycopg2.OperationalError as e:
                self.logger.error(f"数据库连接错误: {e}")
                self.conn.rollback()
                raise
            except psycopg2.Error as e:
                self.logger.error(f"批次导入失败: {e}")
                self.conn.rollback()
                raise
            except Exception as e:
                self.logger.error(f"批次导入失败: {e}")
                self.conn.rollback()
                raise

        try:
            self.conn.commit()
            self.logger.info(f"✅ 数据导入完成: {inserted}行")
        except Exception as e:
            self.logger.error(f"提交事务失败: {e}")
            self.conn.rollback()
            raise

    def _copy_from_dataframe(self, df, table_name):
        """使用COPY命令快速导入（带完整错误处理）"""
        from io import StringIO

        try:
            # 创建CSV字符串
            csv_buffer = StringIO()
            df.to_csv(csv_buffer, index=False, header=False)
            csv_buffer.seek(0)

            # COPY命令
            self.cursor.copy_expert(
                f"COPY {table_name} FROM STDIN WITH CSV",
                csv_buffer
            )
        except psycopg2.OperationalError as e:
            self.logger.error(f"COPY导入失败（连接错误）: {e}")
            raise
        except psycopg2.Error as e:
            self.logger.error(f"COPY导入失败: {e}")
            raise
        except Exception as e:
            self.logger.error(f"COPY导入未知错误: {e}")
            raise

    def create_aggregation_tables(self):
        """创建聚合表（带完整错误处理）"""

        self.logger.info("创建聚合表...")

        try:
            # 日聚合表
            self.cursor.execute("""
                CREATE TABLE IF NOT EXISTS sales_daily AS
                SELECT
                    date_trunc('day', order_date) AS day,
                    product_id,
                    region_id,
                    COUNT(*) AS order_count,
                    SUM(amount) AS total_amount,
                    AVG(amount) AS avg_amount
                FROM sales_fact
                GROUP BY 1, 2, 3;

                CREATE INDEX IF NOT EXISTS idx_sales_daily ON sales_daily (day, product_id, region_id);
            """)

            # 月聚合表
            self.cursor.execute("""
                CREATE TABLE IF NOT EXISTS sales_monthly AS
                SELECT
                    date_trunc('month', day) AS month,
                    product_id,
                    region_id,
                    SUM(order_count) AS order_count,
                    SUM(total_amount) AS total_amount,
                    AVG(avg_amount) AS avg_amount
                FROM sales_daily
                GROUP BY 1, 2, 3;

                CREATE INDEX IF NOT EXISTS idx_sales_monthly ON sales_monthly (month, product_id, region_id);
            """)

            self.conn.commit()
            self.logger.info("✅ 聚合表创建完成")
        except psycopg2.OperationalError as e:
            self.logger.error(f"数据库连接错误: {e}")
            self.conn.rollback()
            raise
        except psycopg2.Error as e:
            self.logger.error(f"创建聚合表失败: {e}")
            self.conn.rollback()
            raise
        except Exception as e:
            self.logger.error(f"未知错误: {e}")
            self.conn.rollback()
            raise

    def refresh_aggregation_tables(self, start_date, end_date):
        """增量刷新聚合表（带完整错误处理）"""

        self.logger.info(f"增量刷新聚合表: {start_date} 至 {end_date}")

        try:
            # 删除旧数据
            self.cursor.execute("""
                DELETE FROM sales_daily
                WHERE day >= %s AND day < %s;
            """, (start_date, end_date))

            # 重新聚合
            self.cursor.execute("""
                INSERT INTO sales_daily
                SELECT
                    date_trunc('day', order_date) AS day,
                    product_id,
                    region_id,
                    COUNT(*) AS order_count,
                    SUM(amount) AS total_amount,
                    AVG(amount) AS avg_amount
                FROM sales_fact
                WHERE order_date >= %s AND order_date < %s
                GROUP BY 1, 2, 3;
            """, (start_date, end_date))

            self.conn.commit()
            self.logger.info("✅ 聚合表刷新完成")
        except psycopg2.OperationalError as e:
            self.logger.error(f"数据库连接错误: {e}")
            self.conn.rollback()
            raise
        except psycopg2.Error as e:
            self.logger.error(f"刷新聚合表失败: {e}")
            self.conn.rollback()
            raise
        except Exception as e:
            self.logger.error(f"未知错误: {e}")
            self.conn.rollback()
            raise

class ParallelQueryExecutor:
    """并行查询执行器"""

    def __init__(self, conn_str, max_workers=4):
        """初始化并行查询执行器（带完整错误处理）"""
        try:
            self.conn_str = conn_str
            self.max_workers = max_workers
        except Exception as e:
            raise RuntimeError(f"初始化失败: {e}")

    def execute_parallel_queries(self, queries):
        """
        并行执行多个查询

        Args:
            queries: 查询列表
        """
        from concurrent.futures import ThreadPoolExecutor
        import time

        results = []

        with ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            futures = [
                executor.submit(self._execute_single_query, q)
                for q in queries
            ]

            for future in futures:
                results.append(future.result())

        return results

    def _execute_single_query(self, query):
        """执行单个查询（带完整错误处理）"""
        import psycopg2
        import time

        conn = None
        cursor = None
        try:
            conn = psycopg2.connect(self.conn_str)
            cursor = conn.cursor()

            start = time.time()
            cursor.execute(query)
            results = cursor.fetchall()
            duration = (time.time() - start) * 1000

            return {
                'query': query,
                'results': results,
                'duration_ms': duration,
                'success': True
            }
        except psycopg2.OperationalError as e:
            return {
                'query': query,
                'results': [],
                'duration_ms': 0,
                'success': False,
                'error': f'数据库连接失败: {str(e)}'
            }
        except psycopg2.Error as e:
            return {
                'query': query,
                'results': [],
                'duration_ms': 0,
                'success': False,
                'error': f'数据库操作失败: {str(e)}'
            }
        except Exception as e:
            return {
                'query': query,
                'results': [],
                'duration_ms': 0,
                'success': False,
                'error': f'未知错误: {str(e)}'
            }
        finally:
            if cursor:
                cursor.close()
            if conn:
                conn.close()

class IncrementalAggregator:
    """增量聚合管理器"""

    def __init__(self, conn):
        self.conn = conn
        self.cursor = conn.cursor()

    def incremental_update(self, last_update_time):
        """
        增量更新聚合表

        Args:
            last_update_time: 上次更新时间
        """

        import psycopg2

        try:
            # 1. 查找需要更新的数据
            self.cursor.execute("""
                SELECT DISTINCT date_trunc('day', order_date) AS day
                FROM sales_fact
                WHERE updated_at > %s;
            """, (last_update_time,))

            affected_days = [row[0] for row in self.cursor.fetchall()]

            if not affected_days:
                print("没有需要更新的数据")
                return

            # 2. 删除受影响的聚合数据
            self.cursor.execute("""
                DELETE FROM sales_daily
                WHERE day = ANY(%s);
            """, (affected_days,))

            # 3. 重新计算聚合
            self.cursor.execute("""
                INSERT INTO sales_daily
                SELECT
                    date_trunc('day', order_date) AS day,
                    product_id,
                    region_id,
                    COUNT(*) AS order_count,
                    SUM(amount) AS total_amount,
                    AVG(amount) AS avg_amount
                FROM sales_fact
                WHERE date_trunc('day', order_date) = ANY(%s)
                GROUP BY 1, 2, 3;
            """, (affected_days,))

            self.conn.commit()
            print(f"✅ 增量更新完成: {len(affected_days)}天")
        except psycopg2.OperationalError as e:
            print(f"数据库连接错误: {e}")
            self.conn.rollback()
            raise
        except psycopg2.Error as e:
            print(f"增量更新失败: {e}")
            self.conn.rollback()
            raise
        except Exception as e:
            print(f"未知错误: {e}")
            self.conn.rollback()
            raise

                rows / calls AS avg_rows
            FROM pg_stat_statements
            WHERE total_exec_time / calls > %s

            raise

            print(f"未知错误: {e}")
            raise

class OLAPQueryMonitor:
    """OLAP查询性能监控"""

    def __init__(self, conn_str):
        self.conn = psycopg2.connect(conn_str)
        self.cursor = self.conn.cursor()

    def log_query(self, query_text: str, duration_ms: float, rows_returned: int):
        """记录查询日志"""
        self.cursor.execute("""
            INSERT INTO olap_query_logs (
                query_text, duration_ms, rows_returned, created_at
            ) VALUES (%s, %s, %s, NOW())
        """, (query_text[:500], duration_ms, rows_returned))
        self.conn.commit()

    def get_slow_queries(self, min_duration_ms: float = 1000, limit: int = 20):
        """获取慢查询"""
        self.cursor.execute("""
            SELECT
                query_text,
                AVG(duration_ms) AS avg_duration_ms,
                COUNT(*) AS execution_count,
                MAX(duration_ms) AS max_duration_ms
            FROM olap_query_logs
            WHERE created_at > NOW() - INTERVAL '24 hours'
            GROUP BY query_text
            HAVING AVG(duration_ms) > %s
            ORDER BY avg_duration_ms DESC
            LIMIT %s
        """, (min_duration_ms, limit))

        return self.cursor.fetchall()
```
