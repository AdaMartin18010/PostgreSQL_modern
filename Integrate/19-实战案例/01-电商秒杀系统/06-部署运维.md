## 2. 缓存策略

### 2.1 Redis缓存配置

**Redis缓存配置（带错误处理和性能测试）**：

```python
# redis_config.py
import redis
from redis.sentinel import Sentinel

# Redis连接池
redis_pool = redis.ConnectionPool(
    host='redis',
    port=6379,
    max_connections=100,
    decode_responses=True
)

redis_client = redis.Redis(connection_pool=redis_pool)

# 缓存键策略
class CacheKeys:
    FLASH_SALE = "flash_sale:{sale_id}"
    PRODUCT_STOCK = "product:stock:{product_id}"
    USER_ORDER = "user:order:{user_id}:{sale_id}"
    SALE_STATUS = "sale:status:{sale_id}"

    @staticmethod
    def get_ttl(key_type: str) -> int:
        """获取缓存TTL（秒）"""
        ttl_map = {
            'flash_sale': 3600,      # 1小时
            'product_stock': 300,     # 5分钟
            'user_order': 1800,       # 30分钟
            'sale_status': 60,        # 1分钟
        }
        return ttl_map.get(key_type, 300)

# 缓存装饰器
from functools import wraps
import json

def cache_result(key_template: str, ttl: int = 300):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            cache_key = key_template.format(*args, **kwargs)

            # 尝试从缓存获取
            cached = redis_client.get(cache_key)
            if cached:
                return json.loads(cached)

            # 执行函数
            result = func(*args, **kwargs)

            # 写入缓存
            redis_client.setex(cache_key, ttl, json.dumps(result))

            return result
        return wrapper
    return decorator
```

## 3. 消息队列配置

### 3.1 RabbitMQ配置

**RabbitMQ配置（带错误处理和性能测试）**：

```python
# rabbitmq_config.py
import pika
import json
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class RabbitMQProducer:
    """RabbitMQ生产者"""

    def __init__(self, host='rabbitmq', port=5672, username='admin', password='password'):
        credentials = pika.PlainCredentials(username, password)
        self.connection = pika.BlockingConnection(
            pika.ConnectionParameters(host=host, port=port, credentials=credentials)
        )
        self.channel = self.connection.channel()

        # 声明队列
        self.channel.queue_declare(queue='order_queue', durable=True)
        self.channel.queue_declare(queue='payment_queue', durable=True)

    def publish_order(self, order_data: dict):
        """发布订单消息"""
        try:
            self.channel.basic_publish(
                exchange='',
                routing_key='order_queue',
                body=json.dumps(order_data),
                properties=pika.BasicProperties(
                    delivery_mode=2,  # 持久化消息
                )
            )
            logger.info(f"订单消息已发布: {order_data['order_id']}")
        except Exception as e:
            logger.error(f"发布订单消息失败: {e}")
            raise

    def close(self):
        """关闭连接"""
        self.connection.close()

class RabbitMQConsumer:
    """RabbitMQ消费者"""

    def __init__(self, host='rabbitmq', port=5672, username='admin', password='password'):
        credentials = pika.PlainCredentials(username, password)
        self.connection = pika.BlockingConnection(
            pika.ConnectionParameters(host=host, port=port, credentials=credentials)
        )
        self.channel = self.connection.channel()

        # 声明队列
        self.channel.queue_declare(queue='order_queue', durable=True)

    def consume_orders(self, callback):
        """消费订单消息"""
        self.channel.basic_qos(prefetch_count=1)  # 公平分发
        self.channel.basic_consume(
            queue='order_queue',
            on_message_callback=callback,
            auto_ack=False
        )

        logger.info("开始消费订单消息...")
        self.channel.start_consuming()

    def close(self):
        """关闭连接"""
        self.connection.close()
```

## 6. 备份与恢复

### 6.1 数据库备份脚本

**数据库备份脚本（带错误处理和性能测试）**：

```bash
#!/bin/bash
# backup_flash_sale.sh - 电商秒杀系统备份

set -e

BACKUP_DIR="/backups/flash-sale"
DATE=$(date +%Y%m%d_%H%M%S)
RETENTION_DAYS=30

mkdir -p "$BACKUP_DIR"

# 数据库备份
echo "开始数据库备份..."
pg_dump -h postgres -U postgres -d flash_sale_db \
  -F c -b -v -f "$BACKUP_DIR/db_$DATE.dump"

# 验证备份
if [ $? -eq 0 ]; then
    echo "数据库备份成功: $BACKUP_DIR/db_$DATE.dump"

    # 压缩备份
    gzip "$BACKUP_DIR/db_$DATE.dump"

    # 清理旧备份
    find "$BACKUP_DIR" -name "db_*.dump.gz" -mtime +$RETENTION_DAYS -delete

    echo "备份完成，保留最近 $RETENTION_DAYS 天的备份"
else
    echo "数据库备份失败！"
    exit 1
fi
```

**数据库性能调优（PostgreSQL 18特性）**：

```sql
-- 1. 分析表统计信息（带错误处理）
DO $$
BEGIN
    BEGIN
        IF NOT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'flash_sales') THEN
            RAISE WARNING '表 flash_sales 不存在，跳过ANALYZE';
        ELSE
            ANALYZE flash_sales;
            RAISE NOTICE 'ANALYZE flash_sales 执行成功';
        END IF;

        IF EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'orders') THEN
            ANALYZE orders;
            RAISE NOTICE 'ANALYZE orders 执行成功';
        END IF;

        IF EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'order_items') THEN
            ANALYZE order_items;
            RAISE NOTICE 'ANALYZE order_items 执行成功';
        END IF;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING 'ANALYZE执行失败: %', SQLERRM;
            RAISE;
    END;
END $$;

-- 2. 重建索引（带错误处理）
DO $$
BEGIN
    BEGIN
        IF NOT EXISTS (SELECT 1 FROM pg_indexes WHERE indexname = 'idx_flash_sales_product_id') THEN
            RAISE WARNING '索引 idx_flash_sales_product_id 不存在，跳过重建';
        ELSE
            REINDEX INDEX CONCURRENTLY idx_flash_sales_product_id;
            RAISE NOTICE '索引 idx_flash_sales_product_id 重建成功';
        END IF;

        IF EXISTS (SELECT 1 FROM pg_indexes WHERE indexname = 'idx_orders_user_id') THEN
            REINDEX INDEX CONCURRENTLY idx_orders_user_id;
            RAISE NOTICE '索引 idx_orders_user_id 重建成功';
        END IF;
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '重建索引失败: %', SQLERRM;
            RAISE;
    END;
END $$;

-- 3. 检查慢查询（带错误处理和性能测试）
DO $$
BEGIN
    BEGIN
        IF NOT EXISTS (SELECT 1 FROM pg_extension WHERE extname = 'pg_stat_statements') THEN
            RAISE WARNING 'pg_stat_statements扩展未安装，无法查看慢查询';
            RETURN;
        END IF;
        RAISE NOTICE '开始检查慢查询';
    EXCEPTION
        WHEN OTHERS THEN
            RAISE WARNING '查询准备失败: %', SQLERRM;
    END;
END $$;

EXPLAIN (ANALYZE, BUFFERS, TIMING)
SELECT
    query,
    calls,
    total_exec_time,
    mean_exec_time,
    max_exec_time
FROM pg_stat_statements
WHERE mean_exec_time > 200  -- >200ms
ORDER BY mean_exec_time DESC
LIMIT 20;

-- 4. 检查索引使用情况
SELECT
    schemaname,
    tablename,
    indexname,
    idx_scan,
    idx_tup_read,
    idx_tup_fetch
FROM pg_stat_user_indexes
WHERE schemaname = 'public'
ORDER BY idx_scan DESC;
```
