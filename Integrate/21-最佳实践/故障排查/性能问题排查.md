WITH vector_search AS (

    FROM documents

),
rrf_fusion AS (

        COALESCE(v.id, t.id) as id,
        COALESCE(v.content, t.content) as content,

    FROM vector_search v
    FULL OUTER JOIN text_search t ON v.id = t.id
)
SELECT id, content, rrf_score
FROM rrf_fusion
ORDER BY rrf_score DESC
LIMIT 10;
