## 摘要

PostgreSQL 18引入了对OAuth2.0和OpenID Connect的原生支持，使得数据库能够直接集成现代身份认证体系。
本文深入分析其实现机制、架构设计和最佳实践。

### 2.2 认证流程实现

```sql
-- JWT验证函数
CREATE OR REPLACE FUNCTION auth.verify_jwt(
    p_token TEXT,
    p_jwks JSONB
) RETURNS TABLE (
    is_valid BOOLEAN,
    username TEXT,
    roles TEXT[],
    claims JSONB
)
AS $$
DECLARE
    v_header JSONB;
    v_payload JSONB;
    v_signature TEXT;
    v_alg TEXT;
    v_kid TEXT;
    v_public_key TEXT;
BEGIN
    -- 解析JWT
    v_header := decode(split_part(p_token, '.', 1), 'base64')::JSONB;
    v_payload := decode(split_part(p_token, '.', 2), 'base64')::JSONB;
    v_signature := split_part(p_token, '.', 3);

    v_alg := v_header->>'alg';
    v_kid := v_header->>'kid';

    -- 获取公钥
    SELECT key INTO v_public_key
    FROM jsonb_array_elements(p_jwks->'keys') AS keys
    WHERE keys->>'kid' = v_kid;

    -- 验证签名
    IF NOT auth.verify_signature(v_alg, v_public_key,
        split_part(p_token, '.', 1) || '.' || split_part(p_token, '.', 2),
        v_signature) THEN
        RETURN QUERY SELECT false, NULL::TEXT, NULL::TEXT[], NULL::JSONB;
        RETURN;
    END IF;

    -- 检查过期
    IF (v_payload->>'exp')::BIGINT < extract(epoch from now()) THEN
        RETURN QUERY SELECT false, NULL::TEXT, NULL::TEXT[], NULL::JSONB;
        RETURN;
    END IF;

    -- 返回验证结果
    RETURN QUERY SELECT
        true,
        v_payload->>'sub',
        ARRAY(SELECT jsonb_array_elements_text(v_payload->'roles')),
        v_payload;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```

## 3. 企业集成架构

### 3.1 Keycloak集成

```yaml
# keycloak配置
keycloak:
  realm: production
  auth-server-url: https://keycloak.example.com
  resource: postgres-client
  credentials:
    secret: ${KEYCLOAK_SECRET}

# PostgreSQL配置
postgresql:
  oauth2:
    issuer: https://keycloak.example.com/realms/production
    jwks_url: https://keycloak.example.com/realms/production/protocol/openid-connect/certs
    role_mapping:
      admin: postgres_admin
      developer: postgres_write
      readonly: postgres_read
```

### 3.2 与LDAP/AD集成

```sql
-- LDAP用户同步
CREATE OR REPLACE PROCEDURE auth.sync_ldap_users()
AS $$
DECLARE
    v_ldap_connection TEXT;
    v_user RECORD;
BEGIN
    -- 连接LDAP
    v_ldap_connection := ldap_connect(
        'ldap://ad.example.com',
        'cn=admin,dc=example,dc=com',
        '${LDAP_PASSWORD}'
    );

    -- 同步用户
    FOR v_user IN
        SELECT * FROM ldap_search(
            v_ldap_connection,
            'ou=users,dc=example,dc=com',
            '(objectClass=person)'
        )
    LOOP
        INSERT INTO auth.users (username, email, ldap_dn)
        VALUES (v_user.cn, v_user.mail, v_user.dn)
        ON CONFLICT (username) DO UPDATE SET
            email = EXCLUDED.email,
            ldap_dn = EXCLUDED.ldap_dn,
            updated_at = NOW();
    END LOOP;
END;
$$ LANGUAGE plpgsql;
```

## 4. 安全最佳实践

### 4.1 令牌刷新机制

```sql
-- 刷新令牌表
CREATE TABLE auth.refresh_tokens (
    token_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id BIGINT REFERENCES auth.users(id),
    token_hash TEXT NOT NULL,
    expires_at TIMESTAMP NOT NULL,
    created_at TIMESTAMP DEFAULT NOW(),
    revoked BOOLEAN DEFAULT false
);

-- 刷新令牌过程
CREATE OR REPLACE FUNCTION auth.refresh_access_token(
    p_refresh_token TEXT
) RETURNS TABLE (
    access_token TEXT,
    expires_in INT
)
AS $$
DECLARE
    v_token_record RECORD;
    v_user RECORD;
BEGIN
    -- 查找刷新令牌
    SELECT * INTO v_token_record
    FROM auth.refresh_tokens
    WHERE token_hash = crypt(p_refresh_token, token_hash)
      AND expires_at > NOW()
      AND NOT revoked;

    IF NOT FOUND THEN
        RAISE EXCEPTION 'Invalid or expired refresh token';
    END IF;

    -- 获取用户信息
    SELECT * INTO v_user FROM auth.users WHERE id = v_token_record.user_id;

    -- 生成新访问令牌
    RETURN QUERY SELECT
        auth.generate_jwt(v_user),
        3600;  -- 1小时有效期
END;
$$ LANGUAGE plpgsql;
```

### 4.2 审计日志

```sql
-- 认证审计表
CREATE TABLE auth.audit_logs (
    id BIGSERIAL PRIMARY KEY,
    event_type VARCHAR(50),  -- login, logout, token_refresh
    username TEXT,
    client_ip INET,
    user_agent TEXT,
    success BOOLEAN,
    error_message TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- 审计触发器
CREATE OR REPLACE FUNCTION auth.log_auth_event()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO auth.audit_logs (event_type, username, success)
    VALUES (TG_OP, NEW.username, true);
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;
```
